  3 /**
  4  * @methodOf       JKIL.sortings.array
  5  * 
  6  * @description    Performs an <b>in-place</b> comparison sort of the given array using an <b>insertion sort algorithm</b> (worst and average performances equal O(<i>n</i><sup>2</sup>), best performance equals O(<i>n</i>)).<br />
  7  * 
  8  * @example        <b>Syntax:</b>
  9  *                 JKIL.sortings.array.doInsertionSort ( pa_array[, ph_options] )
 10  * 
 11  * @example        <b>Example 1:</b>
 12  *                 var a = JKIL.sortings.array.doInsertionSort ( numberArray );
 13  * 
 14  * @example        <b>Example 2:</b>
 15  *                 var a = JKIL.sortings.array.doInsertionSort ( numberArray, {ascending: false} );
 16  * 
 17  * @example        <b>Example 3:</b>
 18  *                 var a = JKIL.sortings.array.doInsertionSort ( numberArray, {ascending: true, comparator: JKIL.comparators.compareNumbers} );
 19  * 
 20  * @param          {Array}       pa_array        The array to be sorted.<br /><br />
 21  * @param          {hash}        [ph_options]    A hash of options:<br /><br />
 22  * 
 23  * @param          {boolean}     [ph_options.ascending=true]               If <code>true</code>, the array is sorted in an ascending manner. If <code>false</code>, the array is sorted in a descending manner.<br /><br />
 24  * @param          {Function}    [ph_options.comparator=compareNumbers]    A callback anonymous function or a reference to a function that manages the comparison between each pair of values. By default, comparison is made with the <code><</code> and <code>></code> operators. 
 25  * 
 26  * @returns        {Array}       The sorted given array (that is, <code>pa_array</code> is modified).
 27  * 
 28  * @throws         TODO
 29  * 
 30  * @see            Rohaut. S. (2007) <i>Algorithmique - Techniques fondamentales de programmation (avec des exemples en Java)</i>, Eni éditions. c. Le tri par sélection p. 112<br /><br />
 31  * @see            http://en.wikipedia.org/wiki/Insertion_sort<br /><br />
 32  * 
 33  * TODO clean
 34  * - https://developer.mozilla.org/fr/docs/Web/JavaScript/Reference/Objets_globaux/Array/sort
 35  * - http://www.javascriptkit.com/javatutors/arraysort.shtml
 36  * - http://www.javascriptkit.com/javatutors/arraysort2.shtml
 37  */
 38 JKIL.sortings.array.doInsertionSort = function (pa_array, ph_options) {
 39     //TODO JKIL.types.isArray( pa_array )
 40     
 41     if (typeof ph_options !== "object" || ph_options === null) {ph_options = {};}
 42     if (typeof ph_options.ascending !== "boolean") {ph_options.ascending = true;}
 43     if (typeof ph_options.comparator !== "function") {
 44         ph_options.comparator = function (pm_a, pm_b) {
 45             if (pm_a < pm_b) {return -1;}
 46             else if (pm_a > pm_b) {return 1}
 47             else {return 0;}
 48         };
 49     }
 50     
 51     var _i_arrayLength = pa_array.length;
 52     var _m_tmp, j;
 53     
 54     for (i=1; i<_i_arrayLength; ++i) {
 55         _m_tmp = pa_array[i];
 56         j = i-1;
 57         
 58         while (    j >= 0
 59                 && (    ph_options.ascending === true && ph_options.comparator(_m_tmp, pa_array[j]) < 0
 60                     ||    ph_options.ascending === false && ph_options.comparator(_m_tmp, pa_array[j]) > 0)
 61                    ) {
 62             pa_array[j+1] = pa_array[j];
 63             --j;
 64         }
 65         pa_array[j+1] = _m_tmp;
 66     }
 67 
 68     return pa_array;
 69 };
