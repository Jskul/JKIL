  3 /**
  4  * @methodOf       JKIL.sortings.array
  5  * 
  6  * @description    Checks whether the given array is sorted or not.<br />
  7  * 
  8  * @example        <b>Syntax:</b>
  9  *                 JKIL.sortings.array.isSorted ( pa_array[, ph_options] )
 10  * 
 11  * @example        <b>Example 1:</b>
 12  *                 var b = JKIL.sortings.array.isSorted ( numberArray );
 13  * 
 14  * @example        <b>Example 2:</b>
 15  *                 var b = JKIL.sortings.array.isSorted ( numberArray, {ascending: null} );
 16  * 
 17  * @example        <b>Example 3:</b>
 18  *                 var b = JKIL.sortings.array.isSorted ( numberArray, {ascending: false} );
 19  *  
 20  * @example        <b>Example 4:</b>
 21  *                 var b = JKIL.sortings.array.isSorted ( numberArray, {ascending: true, comparator: JKIL.comparators.compareNumbers} );
 22  * 
 23  * @param          {Array}           pa_array        The array to be checked.<br /><br />
 24  * @param          {hash}            [ph_options]    A hash of options:<br /><br />
 25  * 
 26  * @param          {null|boolean}    [ph_options.ascending=null]               If <code>null</code>, the method only checks whether the array is sorted. If <code>true</code>, the method checks if the array is sorted in an ascending manner. If <code>false</code>, it checks if the array is sorted in a descending manner.<br /><br />
 27  * @param          {Function}        [ph_options.comparator=compareNumbers]    A callback anonymous function or a reference to a function that manages the comparison between each pair of values. By default, comparison is made with the <code><</code> and <code>></code> operators. 
 28  * 
 29  * @returns        {boolean}         If <code>true</code> the array is sorted, if <code>false</code> the array is not sorted <strong>or</strong> is not sorted according to the given <code>ph_options.ascending</code> boolean value.<br />
 30  *                                   Notice: Arrays having less than two elements are considered sorted.
 31  * 
 32  * @throws      TODO
 33  */
 34 JKIL.sortings.array.isSorted = function (pa_array, ph_options) {
 35     //TODO JKIL.types.isArray( pa_array )
 36     
 37     if (typeof ph_options !== "object" || ph_options === null) {ph_options = {};}
 38     if (typeof ph_options.ascending !== "boolean" && ph_options.ascending !== null) {ph_options.ascending = null;}
 39     if (typeof ph_options.comparator !== "function") {
 40         ph_options.comparator = function (pm_a, pm_b) {
 41             if (pm_a < pm_b) {return -1;}
 42             else if (pm_a > pm_b) {return 1}
 43             else {return 0;}
 44         };
 45     }
 46 
 47     var _b_isSorted = true;
 48     var _i_initialOrder = 0;
 49     var _n_comparison = 0;
 50 
 51     // Arrays having less than two elements are considered sorted.
 52     if ( pa_array.length >= 2 ) {
 53         for ( i=0; _b_isSorted === true && i<pa_array.length - 1; ++i ) {
 54             // Determine the initial ordering of elements according to the two first different elements.
 55             if ( _i_initialOrder === 0 ) {
 56                 _n_comparison = ph_options.comparator(pa_array[i], pa_array[i+1]);
 57                 if ( _n_comparison < 0 ) {
 58                     _i_initialOrder = -1; // The initial ordering is ascending
 59                 } else if ( _n_comparison > 0 ) {
 60                     _i_initialOrder = 1; // The initial ordering is descending
 61                 }
 62                 // If the initial ordering is set, then check if it matches the expected ordering (if any).
 63                 if (    _i_initialOrder === -1 && ph_options.ascending === false
 64                      || _i_initialOrder === 1 && ph_options.ascending === true
 65                    ) {
 66                      _b_isSorted = false;
 67                 }
 68             }
 69              
 70             // Once initial order has been set...
 71             if ( _i_initialOrder !== 0 ) {
 72                 _n_comparison = ph_options.comparator(pa_array[i], pa_array[i+1]);
 73                 // The array is not sorted if at least one pa_array[i] vs. pa_array[i + 1] ordering does not match the initial ordering.
 74                 if (    _i_initialOrder === -1 && _n_comparison > 0
 75                      || _i_initialOrder === 1 && _n_comparison < 0
 76                    ) {
 77                     _b_isSorted = false;
 78                 }
 79             }
 80         }
 81     }
 82     
 83     return _b_isSorted;
 84 };
